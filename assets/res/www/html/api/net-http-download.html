API

M.net.http.download('https://s3-ap-northeast-1.amazonaws.com/morpheus-storage/lab/documents/morpheus_2.0_introduce_v1.3.modified.pdf', {
    overwrite: true,
	progress: function(total, current) {
		console.log(total, current);
	},
	finish: function( statusCode, header, fileInfo ) {
        console.log( statusCode, header, fileInfo );
        
        if (statusCode == 200) {
            alert( JSON.stringify( fileInfo ) );
		}
		else {
            alert( JSON.stringify( arguments ) );
		}
	}
});
